#include <Arduino.h>
#include <WiFi.h>

void setup() {
  Serial.begin(115200);
  
  // Turn on the WiFi chip just enough to read its factory ID
  WiFi.mode(WIFI_MODE_STA);
  
  Serial.println("\n-----------------------------------");
  Serial.print("LA DIRECCIÓN MAC DE ESTA PLACA ES: ");
  Serial.println(WiFi.macAddress());
  Serial.println("-----------------------------------\n");
}

void loop() {
  // Do nothing. Just wait.
  delay(1000);
}
